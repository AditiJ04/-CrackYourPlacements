class Solution {
public:
    bool isOneBitCharacter(vector<int>& bits) {
        int n=bits.size();

        bool set=false;
        for(int i=0;i<n;)
        {
            if(bits[i]==0)
            {
                set=true;
                i++;
            }
            else
            {
                i+=2;
                set=false;
            }
        }

        return set;
    }
};
