class Solution {
public:
    int minOperations(vector<int>& nums) {
        int n=nums.size();

        int flg=0;
        for(int i=1;i<n;i++)
        {
            if(nums[i]!=nums[i-1]) 
            {
                flg=1;
                break;
            }
        }
        if(flg==0) return 0;

        return 1;
    }
};
